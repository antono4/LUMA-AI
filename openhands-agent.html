from fastapi import FastAPI, Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from pydantic import BaseModel
from datetime import datetime, timedelta
import jwt
import passlib.hash as hash

app = FastAPI()
security = HTTPBearer()

# User model
class User(BaseModel):
    username: str
    email: str
    password: str

# Mock database
users_db = []

SECRET_KEY = "your-secret-key-here"
ALGORITHM = "HS256"

def create_token(data: dict):
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(days=7)
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

@app.post("/register")
async def register(user: User):
    if any(u.username == user.username for u in users_db):
        raise HTTPException(status_code=400, detail="Username exists")
    hashed_pw = hash.bcrypt.hash(user.password)
    users_db.append({**user.dict(), "password": hashed_pw})
    return {"message": "User created successfully"}

@app.post("/login")
async def login(user: User):
    db_user = next((u for u in users_db if u["username"] == user.username), None)
    if not db_user or not hash.bcrypt.verify(user.password, db_user["password"]):
        raise HTTPException(status_code=401, detail="Invalid credentials")
    token = create_token({"sub": user.username})
    return {"access_token": token, "token_type": "bearer"}

@app.get("/protected")
async def protected(credentials: HTTPAuthorizationCredentials = Depends(security)):
    token = credentials.credentials
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        return {"user": payload.get("sub"), "status": "authenticated"}
    except:
        raise HTTPException(status_code=401, detail="Invalid token")
